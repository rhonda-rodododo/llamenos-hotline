# Blast Content Encryption Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Encrypt blast content at rest using ECIES envelope pattern, matching the security model of notes and messages.

**Architecture:** Client encrypts on create/edit, server stores opaque ciphertext + ECIES envelopes. Blast processor decrypts per-blast using server's HKDF-derived key. Admin UI decrypts client-side.

**Tech Stack:** @noble/curves, @noble/ciphers, @noble/hashes, Drizzle ORM, bun:test

**Spec:** `docs/superpowers/specs/2026-03-26-blast-content-encryption-design.md`

// --- Blast Content Encryption ---
// Same envelope pattern as messages, using LABEL_BLAST_CONTENT for domain separation.
// Cannot reuse encryptMessage because it hardcodes LABEL_MESSAGE.

import { LABEL_BLAST_CONTENT } from '@shared/crypto-labels'
// (Add LABEL_BLAST_CONTENT to the existing import at line 8)

export interface EncryptedBlastContentPayload {
  encryptedContent: string // hex: nonce(24) + ciphertext
  contentEnvelopes: RecipientKeyEnvelope[] // blast key wrapped for each admin + server
}

/**
 * Encrypt blast content for storage using the envelope pattern.
 * Generates a random per-blast symmetric key, wraps it for each recipient
 * (admins + server pubkey) via ECIES with LABEL_BLAST_CONTENT.
 *
 * @param content - BlastContent object to encrypt (JSON-serialized)
 * @param recipientPubkeys - Array of x-only pubkeys (admin(s) + server)
 */
export function encryptBlastContent(
  content: BlastContent,
  recipientPubkeys: string[]
): EncryptedBlastContentPayload {
  const blastKey = randomBytes(32)
  const nonce = randomBytes(24)
  const cipher = xchacha20poly1305(blastKey, nonce)
  const ciphertext = cipher.encrypt(utf8ToBytes(JSON.stringify(content)))

  const packed = new Uint8Array(nonce.length + ciphertext.length)
  packed.set(nonce)
  packed.set(ciphertext, nonce.length)

  return {
    encryptedContent: bytesToHex(packed),
    contentEnvelopes: recipientPubkeys.map((pk) => ({
      pubkey: pk,
      ...eciesWrapKey(blastKey, pk, LABEL_BLAST_CONTENT),
    })),
  }
}

/**
 * Decrypt blast content from an ECIES envelope.
 * Finds the envelope matching the reader's pubkey and unwraps.
 *
 * @param encryptedContent - hex: nonce(24) + ciphertext
 * @param contentEnvelopes - array of per-recipient ECIES envelopes
 * @param secretKey - reader's secret key (Uint8Array)
 * @param readerPubkey - reader's x-only pubkey (hex)
 */
export function decryptBlastContent(
  encryptedContent: string,
  contentEnvelopes: RecipientKeyEnvelope[],
  secretKey: Uint8Array,
  readerPubkey: string
): BlastContent | null {
  try {
    const envelope = contentEnvelopes.find((e) => e.pubkey === readerPubkey)
    if (!envelope) return null

    const blastKey = eciesUnwrapKey(envelope, secretKey, LABEL_BLAST_CONTENT)

    const data = hexToBytes(encryptedContent)
    const nonce = data.slice(0, 24)
    const ciphertext = data.slice(24)
    const cipher = xchacha20poly1305(blastKey, nonce)
    const plaintext = cipher.decrypt(ciphertext)
    return JSON.parse(new TextDecoder().decode(plaintext)) as BlastContent
  } catch {
    return null
  }
}
```

```

And add `LABEL_BLAST_CONTENT` to the existing crypto-labels import on line 8.

   * Decrypt blast content using the server's ECIES envelope.
   * Derives the server private key from SERVER_NOSTR_SECRET via HKDF,
   * finds the server's envelope in contentEnvelopes, unwraps the blast key,
   * and decrypts the content. Override in tests.

  async _decryptBlastContent(blast: Blast): Promise<string> {
    const { secretKey, pubkey: serverPubkey } = deriveServerKeypair(this.serverSecret)
    const envelope = blast.contentEnvelopes.find((e) => e.pubkey === serverPubkey)
    if (!envelope) {
      throw new Error('No content envelope for server pubkey')

    const blastKey = eciesUnwrapKeyServer(envelope, secretKey, LABEL_BLAST_CONTENT)

    const packed = hexToBytes(blast.encryptedContent)
    const nonce = packed.slice(0, 24)
    const ciphertext = packed.slice(24)
    const cipher = xchacha20poly1305(blastKey, nonce)
    const plaintext = cipher.decrypt(ciphertext)
    const content = JSON.parse(new TextDecoder().decode(plaintext))
    return content.text
  }

import { describe, expect, test } from 'bun:test'
import { getPublicKey } from 'nostr-tools'
import { LABEL_BLAST_CONTENT, LABEL_MESSAGE } from '@shared/crypto-labels'
import type { BlastContent } from '@shared/types'
import {
  decryptBlastContent,
  decryptMessage,
  eciesUnwrapKey,
  eciesWrapKey,
  encryptBlastContent,
  encryptMessage,
} from './crypto'

function makeKeypair() {
  const sk = new Uint8Array(32)
  crypto.getRandomValues(sk)
  // Ensure valid secp256k1 scalar
  sk[0] = sk[0] & 0x7f
  if (sk.every((b) => b === 0)) sk[31] = 1
  const pk = getPublicKey(sk)
  return { sk, pk }
}

describe('blast content encryption', () => {
  const admin1 = makeKeypair()
  const admin2 = makeKeypair()
  const server = makeKeypair()

  const content: BlastContent = {
    text: 'Hello subscribers!',
    mediaUrl: 'https://example.com/image.png',
    smsText: 'Hello via SMS!',
  }

  test('encrypt → decrypt roundtrip with admin key', () => {
    const { encryptedContent, contentEnvelopes } = encryptBlastContent(content, [
      admin1.pk,
      server.pk,
    ])

    expect(encryptedContent).toBeTruthy()
    expect(contentEnvelopes).toHaveLength(2)

    const decrypted = decryptBlastContent(encryptedContent, contentEnvelopes, admin1.sk, admin1.pk)
    expect(decrypted).toEqual(content)
  })

  test('encrypt → decrypt roundtrip with server key', () => {
    const { encryptedContent, contentEnvelopes } = encryptBlastContent(content, [
      admin1.pk,
      server.pk,
    ])

    const decrypted = decryptBlastContent(encryptedContent, contentEnvelopes, server.sk, server.pk)
    expect(decrypted).toEqual(content)
  })

  test('multi-admin: both admins can decrypt', () => {
    const { encryptedContent, contentEnvelopes } = encryptBlastContent(content, [
      admin1.pk,
      admin2.pk,
      server.pk,
    ])

    expect(contentEnvelopes).toHaveLength(3)

    const d1 = decryptBlastContent(encryptedContent, contentEnvelopes, admin1.sk, admin1.pk)
    const d2 = decryptBlastContent(encryptedContent, contentEnvelopes, admin2.sk, admin2.pk)
    expect(d1).toEqual(content)
    expect(d2).toEqual(content)
  })

  test('wrong key returns null', () => {
    const { encryptedContent, contentEnvelopes } = encryptBlastContent(content, [
      admin1.pk,
      server.pk,
    ])

    const wrongKey = makeKeypair()
    const decrypted = decryptBlastContent(
      encryptedContent,
      contentEnvelopes,
      wrongKey.sk,
      wrongKey.pk
    )
    expect(decrypted).toBeNull()
  })

  test('domain separation: LABEL_BLAST_CONTENT vs LABEL_MESSAGE are incompatible', () => {
    // Encrypt a message with LABEL_MESSAGE
    const { encryptedContent, readerEnvelopes } = encryptMessage('hello', [admin1.pk])

    // Try to decrypt as blast content (wrong label)

      readerEnvelopes,
      admin1.sk,
      admin1.pk
    )
    expect(decrypted).toBeNull()
  })

  test('empty content envelopes returns null', () => {
    const { encryptedContent } = encryptBlastContent(content, [admin1.pk])
    const decrypted = decryptBlastContent(encryptedContent, [], admin1.sk, admin1.pk)
    expect(decrypted).toBeNull()
  })
})
